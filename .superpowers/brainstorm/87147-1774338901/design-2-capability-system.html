CapabilityProfile - Output Type
interface CapabilityProfile {
  primary: CapabilityTag[]; // ["reasoning", "code-gen"]
  preference: "speed" | "cost" | "quality";
  confidence: number; // 0-1, heuristic high=0.9, LLM=0.7
  source: "heuristic" | "llm-triage" | "hybrid";
}
 
type CapabilityTag =
  | "reasoning" | "vision" | "code-gen"
  | "tool-use" | "long-context"
  | "speed" | "cost" | "quality" | "creative";
 
interface TaggedGoalNode extends GoalNode {
  capabilityProfile: CapabilityProfile;
  assignedProvider?: string; // ProviderAssigner tarafindan eklenir
  assignedModel?: string;
}
